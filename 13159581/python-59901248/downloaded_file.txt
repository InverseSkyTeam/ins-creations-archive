    yscrollbar2.pack(side=RIGHT, fill=Y)
    outtext.pack()

    outtext.config(yscrollcommand=yscrollbar2.set)

    outtext.delete('1.0', 'end')
    outtext['state'] = 'disabled'
    def submit():
        def print(*value, sep=' ', end='\n'):
            outtext['state'] = 'normal'
            for i in range(len(value)):
                outtext.insert('end', value[i])
                if i < len(value) - 1:
                    outtext.insert('end', sep)
                    
            outtext.insert('end', end)
            outtext.see('end')
            time.sleep(0.03)
            outtext['state'] = 'disabled'


        def input(value=''):
            print(value, end='')
            is_function_finished = BooleanVar()
            is_function_finished.set(False)
            submittext = ""

            def on_key_press(event):
                nonlocal is_function_finished
                nonlocal submittext
                text = event.widget
                line, column = map(int, text.index("insert").split('.'))
                current_line_content = text.get("{}.0".format(line), "{}.end".format(line))
                if event.keysym == "BackSpace":
                    if (not current_line_content.strip().replace(value, "")):
                        return "break"
                    else:
                        submittext = submittext[:-1]
                        last_char_index = 'end-1c'
                        outtext.delete(last_char_index)
                elif event.keysym == "Return":
                    is_function_finished.set(True)
                else:
                    submittext += event.char
                    # outtext.insert('end', event.char)

            outtext['state'] = 'normal'
            outtext.bind("<Key>", on_key_press)
            outtext.wait_variable(is_function_finished)
            outtext.unbind("<Key>")
            outtext["state"] = "disabled"
            return submittext
        
        try:
            sys.stdout.write = print
            sys.stdin.readline = input
            exec(code_text, globals(), locals())
        except Exception as b:
            bug = traceback.format_exc()
            try:
                print(bug)
            except:
                pass
    
    threada = threading.Thread(target=submit)
    threada.daemon = True
    threada.start()
    frame3.mainloop()

class TreeWindow(ttk.Frame):
    def __init__(self, master, path):
        super().__init__(master)
        self.pack()

        self.tree = ttk.Treeview(self)
        self.tree.pack(fill=BOTH, expand=True)

        # path
        self.current_path = path

        root = self.tree.insert("", "end", text=self.get_last_path(path), open=True)
        self.load_tree(root, path)


    def load_tree(self, parent, path):
        self.tree.delete(*self.tree.get_children(parent))
        for filepath in os.listdir(path):
            abs_path = os.path.join(path, filepath)
            node_text = self.get_last_path(abs_path)
            node = self.tree.insert(parent, "end", text=node_text)
            if os.path.isdir(abs_path):
                self.load_tree(node, abs_path)

    def get_last_path(self, path):
        return os.path.basename(path)

    def change_path(self):
        new_path = filedialog.askdirectory(initialdir=self.current_path)
        if new_path:
            self.current_path = new_path
            self.tree.delete(*self.tree.get_children())
            root = self.tree.insert("", "end", text=self.get_last_path(new_path), open=True)
            self.load_tree(root, new_path)

    def get_full_path(self, item):
        parent = self.tree.parent(item)
        if parent == "":
            return self.tree.item(item, "text")
        return (self.get_full_path(parent)+'/'+self.tree.item(item, "text")).replace(self.current_path.split("/")[-1],"")

    def configCodeHighlight(self, component: Text):
        idc.color_config(self.text)
        p = idp.Percolator(self.text)
        d = idc.ColorDelegator()
        p.insertfilter(d)

    def on_key_press(self,event):
        text = event.widget
        line, column = map(int, text.index("insert").split('.'))
        current_line_content = text.get("{}.0".format(line), "{}.end".format(line))
        if event.keysym == "Return" and current_line_content!="":
            if (current_line_content[-1]==":"):
                self.text.insert(END,"\n    ")
                return "break"
        if event.keysym == "Tab":
            self.text.insert(END,"    ")
            return "break"

    def save(self):
        try:
            t = self.text.get(1.0, END)
            f = filedialog.asksaveasfile(defaultextension=".py")
            f.write(t + "\ninput(\"按回车退出\")")
        except:
            messagebox.showinfo('提示', '保存失败')
        else:
            messagebox.showinfo('提示', '保存成功')

    def open(self):
        path = filedialog.askopenfilename(initialdir="C:\\")
        if path:
            self.text["state"] = "normal"
            self.tk.bind("<F5>",self.aaaa)
            self.filename = ""
            with open(path, "r", encoding="utf-8") as file:
                self.text.delete("1.0", "end")
                self.text.insert(END, file.read())

    def on_file_selected(self, event):
        selected_item = self.tree_window.tree.selection()[0]
        self.text["state"] = "normal"
        self.text.delete("1.0", "end")
        item_text = self.tree_window.tree.item(selected_item, "text")
        file_extension = os.path.splitext(item_text)[1]
        if file_extension == '.py':
            self.file_menu.entryconfig(0,state=NORMAL)
            self.tk.bind("<F5>",self.aaaa)
            self.text["state"] = "normal"
            selected_file_path = self.tree_window.current_path+self.tree_window.get_full_path(selected_item)
            try:
                with open(self.filename,"w",encoding="utf-8") as f:
                    f.write(self.text.get(1.0,END))
            except Exception as b:
                print(b)
            with open(selected_file_path,"r",encoding="utf-8") as file:
                self.filename = selected_file_path
                self.text.delete("1.0", "end")
                self.text.insert(END, file.read())
            
        elif not file_extension in ('.png', '.jpg', '.jpeg', '.gif', '.bmp', '.webp'):
            self.file_menu.entryconfig(0,state=DISABLED)
            self.tk.unbind("<F5>")
            selected_file_path = self.tree_window.current_path+self.tree_window.get_full_path(selected_item)
            try:
                with open(self.filename,"w",encoding="utf-8") as f:
                    f.write(self.text.get(1.0,END))
                with open(selected_file_path,"r",encoding="utf-8") as file:
                    self.filename = selected_file_path
                    self.text.delete("1.0", "end")
                    self.text.insert(END, file.read())
            except:
                pass
